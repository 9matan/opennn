139 
140  check();
141 
142 #endif

144  GrowingInputsResults* results = new GrowingInputsResults();
145 
146  size_t index;
147 
148  size_t original_index;
149 
150  const PerformanceFunctional* performance_functional_pointer = training_strategy_pointer->get_performance_functional_pointer();
151 
152  NeuralNetwork* neural_network_pointer = performance_functional_pointer->get_neural_network_pointer();
153 
154  DataSet* data_set_pointer = performance_functional_pointer->get_data_set_pointer();
155

157

159 
160  const size_t targets_number = variables->count_targets_number();
161 
162  Vector< Statistics<double> > original_statistics;
163  ScalingLayer::ScalingMethod original_scaling_method;
164 
165  bool has_scaling_layer = neural_network_pointer->has_scaling_layer();
166 
167  if (has_scaling_layer)
168  {
169  original_statistics = neural_network_pointer->get_scaling_layer_pointer()->get_statistics();
170  original_scaling_method = neural_network_pointer->get_scaling_layer_pointer()->get_scaling_method();
171  }
172 
173  Vector<double> current_inputs(inputs_number, 1);
174 
175  Vector<Variables::Use> current_uses = variables->arrange_uses();
176  const Vector<Variables::Use> original_uses = current_uses;
177 
178  double optimum_generalization_error = 1e10;
179  double optimum_performance_error;
180  Vector<double> optimal_inputs;
181  Vector<double> optimal_parameters;
182 
183  Vector<double> final_correlations;
184 
185  Vector<double> final(2);
186  Vector<double> history_row;
187 
188  double current_training_performance, current_generalization_performance;
189  double previous_generalization_performance;
190 
191  bool flag_input = false;
192 
193  bool end = false;
194 
195  size_t iterations = 0;
196 
197  size_t generalization_failures = 0;
198 
199  time_t beginning_time, current_time;
200  double elapsed_time;
201 
202  if (display)
203  {
204  std::cout << "Performing growing inputs selection..." << std::endl;
205  std::cout << std::endl << "Calculating correlations..." << std::endl;
206  }
207 
208  final_correlations = calculate_final_correlations();
209 
210  if (display)
211  {
212  std::cout << "Correlations : \n";
213  for(size_t i = 0; i < final_correlations.size(); i++)
214  {
215  original_index = get_input_index(original_uses, i);
216 
217  std::cout << "input : " << variables->arrange_names()[original_index] << ", correlation = " << final_correlations[i] << std::endl;
218 
219  if (i == 9)
220  {
221  std::cout << "...\n";
222  break;
223  }
224  }
225  }
226 
227  for (size_t i = 0; i < current_uses.size(); i++)
228  if(current_uses[i] == Variables::Input)
229  current_uses[i] = Variables::Unused;
230 
231  current_inputs.set(inputs_number, 0);
232 
233  time(&beginning_time);
234 
235  index = final_correlations.calculate_maximal_index();
236 
237  if (final_correlations[index] >= 0.9999*targets_number)
238  {
239  original_index = get_input_index(original_uses, index);
240 
241  current_uses[original_index] = Variables::Input;
242 
243  current_inputs[index] = 1;
244 
245  variables->set_uses(current_uses);
246 
247  set_neural_inputs(current_inputs);
248 
249  if(display)
250  std::cout << "Maximal correlation(" << final_correlations[index] << ") is nearly 1. \n"
251  << "The problem is linear separable with the input " << variables->arrange_names()[original_index] << ".\n\n" ;
252 
253  final = calculate_performances(current_inputs);
254 
255  optimal_inputs = current_inputs;
256  optimum_generalization_error = final[0];
257  optimum_performance_error = final[1];
258  optimal_parameters = get_parameters_inputs(current_inputs);
259 
260  if (reserve_performance_data)
261  {
262  history_row = current_inputs;
263  history_row.push_back(final[0]);
264  results->performance_data.push_back(history_row);
265  }
266 
267  if (reserve_generalization_performance_data)
268  {
269  history_row = current_inputs;
270  history_row.push_back(final[1]);
271  results->generalization_performance_data.push_back(history_row);
272  }
273 
274  if (reserve_parameters_data)
275  {
276  history_row = get_parameters_inputs(current_inputs);
277  history_row.insert(history_row.begin(),current_inputs.begin(),current_inputs.begin()+current_inputs.size());
278  results->parameters_data.push_back(history_row);
279  }
280 
281  results->stopping_condition = InputsSelectionAlgorithm::AlgorithmFinished;
282 
283  end = true;
284  }else
285  {
286  for(size_t i = 0; i < final_correlations.size(); i++)
287  if(final_correlations[i] >= maximum_correlation*targets_number)
288  {
289  final_correlations[i] = -1;
290 
291  original_index = get_input_index(original_uses, i);
292 
293  current_uses[original_index] = Variables::Input;
294 
295  current_inputs[i] = 1;
296 
297  variables->set_uses(current_uses);
298 
299  set_neural_inputs(current_inputs);
300 
301  if(display)
302  std::cout << "Added input "<< variables->arrange_names()[original_index] << std::endl;
303  flag_input = true;
304  }
305  if(display)
306  std::cout << std::endl;
307 
308  }
309 
310  while(!end)
311  {
312 
313  if(!flag_input)
314  {
315  index = final_correlations.calculate_maximal_index();
316 
317  if(final_correlations[index] <= minimum_correlation*targets_number)
318  {
319  end = true;
320  if (display)
321  std::cout << "Correlation goal reached." << std::endl << std::endl;
322  results->stopping_condition = InputsSelectionAlgorithm::CorrelationGoal;
323  }else
324  {
325 
326  final_correlations[index] = -1;
327 
328  original_index = get_input_index(original_uses, index);
329 
330  current_uses[original_index] = Variables::Input;
331 
332  current_inputs[index] = 1;
333 
334  variables->set_uses(current_uses);
335 
336  set_neural_inputs(current_inputs);
337 
338  }
339  }
340 
341  final = calculate_performances(current_inputs);
342 
343  current_training_performance = final[0];
344  current_generalization_performance = final[1];
345 
346  if (fabs(optimum_generalization_error - current_generalization_performance) < tolerance ||
347  optimum_generalization_error > current_generalization_performance)
348  {
349  optimal_inputs = current_inputs;
350  optimum_performance_error = current_training_performance;
351  optimum_generalization_error = current_generalization_performance;
352  }
353  else if (optimum_generalization_error < current_generalization_performance)
354  //previous_generalization_performance < current_generalization_performance)
355  generalization_failures++;
356 
357  time(&current_time);
358  elapsed_time = difftime(current_time, beginning_time);
359 
360  previous_generalization_performance = current_generalization_performance;
361  iterations++;
362 
363  if (reserve_performance_data)
364  {
365  history_row = current_inputs;
366  history_row.push_back(final[0]);
367  results->performance_data.push_back(history_row);
368  }
369 
370  if (reserve_generalization_performance_data)
371  {
372  history_row = current_inputs;
373  history_row.push_back(final[1]);
374  results->generalization_performance_data.push_back(history_row);
375  }
376 
377  if (reserve_parameters_data)
378  {
379  history_row = get_parameters_inputs(current_inputs);
380  history_row.insert(history_row.begin(),current_inputs.begin(),current_inputs.begin()+current_inputs.size());
381  results->parameters_data.push_back(history_row);
382  }
383 
384  // STOPPING CRITERIA
385 
386  if (elapsed_time > maximum_time)
387  {
388  end = true;
389  if (display)
390  std::cout << "Maximum time reached." << std::endl;
391  results->stopping_condition = InputsSelectionAlgorithm::MaximumTime;
392  }else if (final[1] < generalization_performance_goal)
393  {
394  end = true;
395  if (display)
396  std::cout << "Generalization performance reached." << std::endl;
397  results->stopping_condition = InputsSelectionAlgorithm::GeneralizationPerformanceGoal;
398  }else if (iterations > maximum_iterations_number)
399  {
400  end = true;
401  if (display)
402  std::cout << "Maximum number of iterations reached." << std::endl;
403  results->stopping_condition = InputsSelectionAlgorithm::MaximumIterations;
404  }else if (generalization_failures >= maximum_generalization_failures)
405  {
406  end = true;
407  if (display)
408  std::cout << "Maximum generalization performance failures("<<generalization_failures<<") reached." << std::endl;
409  results->stopping_condition = InputsSelectionAlgorithm::MaximumGeneralizationFailures;
410  }else if (current_inputs.count_occurrences(1) == current_inputs.size())
411  {
412  end = true;
413  if (display)
414  std::cout << "Algorithm finished" << std::endl;
415  results->stopping_condition = InputsSelectionAlgorithm::AlgorithmFinished;
416  }
417 
418  if (display)
419  {
420  std::cout << "Iteration : " << iterations << std::endl;
421  if(!flag_input)
422  std::cout << "Add input : " << variables->arrange_names()[original_index] << std::endl;
423  std::cout << "Current inputs : " << variables->arrange_inputs_name().to_string() << std::endl;
424  std::cout << "Number of inputs : " << current_inputs.count_occurrences(1) << std::endl;
425  std::cout << "Training performance : " << final[0] << std::endl;
426  std::cout << "Generalization error : " << final[1] << std::endl;
427  std::cout << "Elapsed time : " << elapsed_time << std::endl;
428 
429  std::cout << std::endl;
430  }
431  flag_input = false;
432  }
433 
434  optimal_parameters = get_parameters_inputs(optimal_inputs);
435  if (reserve_minimal_parameters)
436  results->minimal_parameters = optimal_parameters;
437 
438 
439  results->optimal_inputs = optimal_inputs;
440  results->final_generalization_performance = optimum_generalization_error;
441  results->final_performance = calculate_performances(optimal_inputs)[0];
442  results->iterations_number = iterations;
443  results->elapsed_time = elapsed_time;
444 
445  Vector< Statistics<double> > statistics;
446  for (size_t i = 0; i < optimal_inputs.size(); i++)
447  {
448  original_index = get_input_index(original_uses, i);
449  if (optimal_inputs[i] == 1)
450  {
451  current_uses[original_index] = Variables::Input;
452  if (has_scaling_layer)
453  statistics.push_back(original_statistics[i]);
454  }else
455  {
456  current_uses[original_index] = Variables::Unused;
457  }
458  }
459 
460  variables->set_uses(current_uses);
461 
462  set_neural_inputs(optimal_inputs);
463  neural_network_pointer->set_parameters(optimal_parameters);
464  neural_network_pointer->get_inputs_pointer()->set_names(variables->arrange_inputs_name());
465 
466  if (has_scaling_layer)
467  {
468  ScalingLayer scaling_layer(statistics);
469  scaling_layer.set_scaling_method(original_scaling_method);
470  neural_network_pointer->set_scaling_layer(scaling_layer);
471  }
472 
473  time(&current_time);
474  elapsed_time = difftime(current_time, beginning_time);
475 
476  if (display)
477  {
478  std::cout << "Optimal inputs : " << neural_network_pointer->get_inputs_pointer()->arrange_names().to_string() << std::endl;
479  std::cout << "Optimal number of inputs : " << optimal_inputs.count_occurrences(1) << std::endl;
480  std::cout << "Optimum training performance : " << optimum_performance_error << std::endl;
481  std::cout << "Optimum generalization error : " << optimum_generalization_error << std::endl;
482  std::cout << "Elapsed time : " << elapsed_time << std::endl;
483  }
484 
485  return results;
486 }
