950 tinyxml2::XMLDocument* Instances::to_XML(void) const
951 {
952  tinyxml2::XMLDocument* document = new tinyxml2::XMLDocument;
953 
954  std::ostringstream buffer;
955 
956  // Instances
957 
958  tinyxml2::XMLElement* instances_element = document->NewElement("Instances");
959 
960  document->InsertFirstChild(instances_element);
961 
962  tinyxml2::XMLElement* element = NULL;
963  tinyxml2::XMLText* text = NULL;
964 
965  const size_t instances_number = get_instances_number();
966 
967  // Instances number
968  {
969  element = document->NewElement("InstancesNumber");
970  instances_element->LinkEndChild(element);
971 
972  buffer.str("");
973  buffer << instances_number;
974 
975  text = document->NewText(buffer.str().c_str());
976  element->LinkEndChild(text);
977  }
978 
979  for(size_t i = 0; i < instances_number; i++)
980  {
981  element = document->NewElement("Item");
982  element->SetAttribute("Index", (unsigned)i+1);
983  instances_element->LinkEndChild(element);
984 
985  // Use
986 
987  tinyxml2::XMLElement* use_element = document->NewElement("Use");
988  element->LinkEndChild(use_element);
